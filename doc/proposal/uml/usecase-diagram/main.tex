\usepackage{tikz} % for tikzpicture
\usetikzlibrary{positioning}

% The actor shape without scope or ports
\newcommand*\unscopedactor{%
    \draw (0,22.5pt) circle (7.5pt);% head
    \draw (0,15pt) -- ++(0,-25pt) -- ++(-15pt,-20pt);% body and left leg
    \draw (-15pt,10pt) -- ++(30pt,0);% arms
    \draw (0,-10pt) -- ++(15pt,-20pt)% right leg
}%

% The actor shape including scope and ports
% @param 1 = scale (default 1)
% @param 2 = port name prefix
% @param 3 = x-shift
% @param 4 = y-shift
% @param 5 = commands after the actor within scope
\newcommand*\actor[5][1]{%
    \begin{scope}[scale=#1,xshift=#3,yshift=#4,line width=1pt]%
        \unscopedactor;%
        % create the actor ports
        \foreach \n/\x/\y in {
            /0/0,
            -north/0/30pt,
            -west/-15pt/10pt,
            -east/15pt/10pt,
            -south/0/-30pt
        }
            \coordinate (#2\n) at (\x,\y);
        % \foreach \n/\x/\y
        #5
    \end{scope}%
}%

\begin{document}

\begin{tikzpicture}
    % actors
    \actor{A}{0}{0};
    \actor{B}{0}{-1.5in};
    \actor{C}{1in}{0};
    % logos
    \node at (A) {A};
    \node at (B) {B};
    \node at (C) {C};
    % labels
    \node[below=0pt of A-south] {actor};
    \node[below=0pt of B-south] {band};
    \node[below=0pt of C-south] {cowboy};
    % relations
    \draw (A-east) -- (C-west);
    \draw (B-east) -- (C-west);
\end{tikzpicture}

\end{document}
